Jasmine

describe('The test suite', function() {
    it('Should do the right thing', function(){
      expect(spikeLee === 'do the right thing.').toBe(true);
    });

    it('Should do some math', function(){
      expect(2+2).toBe(4);
    });
  }
});
  • Behavior-driven framework for testing JavaScript code.
  • No other JavaScript dependencies.
  • Very natural syntax/api.
  • Built by Pivotal Labs

QUnit

test("hello test", function() {
  ok(1 == "1", "Passed!");
});
  • Used to be part of jQuery.
  • QUnit was pulled out into its own tool.
  • Used to test used to test the jQuery library but isn't limited to jQuery code.
  • Simple & easy to use.

FuncUnit

module('ABA Login View', {
    setup: function() {
      S.open('http://aba.realeyes.com/test/#!/login');
    }
  }
});
test('Login View', function() {
  expect(3);
  ok(S('#username_txt').exists(), 'username_txt control should exist.');
  ok(S('#password_txt').exists(), 'username_txt control should exist.');
  ok(S('#submit_btn').exists(), 'username_txt control should exist.');
});
test('Login', function() {
  S('#username_txt').click().type('[shift]A[shift-up]0604283');
  S('#password_txt').click().type('test');
  S('#submit_btn span.ui-btn-text').click();
  ok(S('.centeredContainer').visible());
});
